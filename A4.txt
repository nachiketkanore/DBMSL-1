select * from customer 
natural join orders;

select b.title,a.tuthor_Name,a.country 
from authors a 
inner join books b 
where a.author_No = b.author_No;

select c.customer_id, c.first_name, o.order_id
-> from customer c
-> left join orders o
-> on c.customer_id = o.customer_id
-> where order_id is NULL;


select o.Order_No, b.ISBN, b.Title 
from Orders o
right join Book b
on o.ISBN = b.ISBN 
where Order_No is NULL;

select c.First_Name,b.Title,b.Author_No,b.Year 
from Customer c
natural join Orders o
natural join Book b 
where ISBN = 112233;

select c.First_Name, sum(o.Quantity) as Total_Books
from Customer c
natural join Orders o
group by o.Cust_No;

select o.Cust_No, o.Order_No, o.ISBN 
from Orders o
right join Book
on o.ISBN = Book.ISBN 
where Book.Title = "DSA";

select c.Company 
from Customer c 
natural join Orders 
where Year(ODate) = 2015;
